function isArmstrongNumber(number) { // Convert number to string to iterate through each digit let numStr = number.toString(); // Calculate the number of digits let numberOfDigits = numStr.length; // Initialize sum let sum = 0; // Iterate through each digit for (let digit of numStr) { // Convert digit back to number and calculate digit^numberOfDigits sum += Math.pow(parseInt(digit), numberOfDigits); } // Check if the sum is equal to the original number return sum === number; } // Example usage: let num = 153; if (isArmstrongNumber(num)) { console.log(`${num} is an Armstrong number`); } else { console.log(`${num} is not an Armstrong number`); }
